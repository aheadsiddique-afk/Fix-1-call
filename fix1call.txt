    <script>
        document.getElementById('contactForm').addEventListener('submit', function(e) {
            e.preventDefault();
            const name = this.name.value.trim();
            const mobile = this.mobile.value.trim();
            const messageBox = document.getElementById('messageBox');

            if (name === '' || mobile === '') {
                messageBox.classList.remove('hidden', 'bg-green-100', 'text-green-800', 'bg-red-100', 'text-red-800');
                messageBox.classList.add('bg-red-100', 'text-red-800');
                messageBox.textContent = 'Please fill both Name and Mobile Number.';
            } else if (!/^\d{10}$/.test(mobile)) {
                messageBox.classList.remove('hidden', 'bg-green-100', 'text-green-800', 'bg-red-100', 'text-red-800');
                messageBox.classList.add('bg-red-100', 'text-red-800');
                messageBox.textContent = 'Please enter a valid 10-digit Mobile Number.';
            } else {
                // Here you would typically send the form data to a server
                console.log('Form submitted:', { name, mobile });
                messageBox.classList.remove('hidden', 'bg-green-100', 'text-green-800', 'bg-red-100', 'text-red-800');

                this.reset();
            }
        });
